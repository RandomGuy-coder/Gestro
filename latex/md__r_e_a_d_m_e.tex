\hypertarget{md__r_e_a_d_m_e_autotoc_md1}{}\doxysection{To control the Ubuntu system using hand gestures}\label{md__r_e_a_d_m_e_autotoc_md1}
{\bfseries{Functions\+:}}
\begin{DoxyItemize}
\item Increase or Decrease Volume and Brightness.
\item Switch between applications.
\item Zoom in and Zoom out.
\item Track mouse.
\item Mute and Unmute.
\item Play and Pause.
\item Next and previous.
\end{DoxyItemize}

{\bfseries{Methodologies\+:}}
\begin{DoxyItemize}
\item A PC running Ubuntu.
\item A webcam.
\item Using Open\+CV and Tensor\+Flow.
\end{DoxyItemize}

{\bfseries{Problems Being Solved\+:}}
\begin{DoxyItemize}
\item Additional everyday functionalities for one that is sitting away from their PC (Eg. When giving a presentation).
\item To aid users with poor fine motor control (Eg. Parkinson\textquotesingle{}s patients).
\item To make controlling the system more user friendly for people that are unfamiliar with modern devices (Eg. Elderly people)
\end{DoxyItemize}
